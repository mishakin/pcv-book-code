cg_similarity_2D

PURPOSE

% determines constraint values and Jacobians for 2D similarity

SYNOPSIS

function [cg,At,Bt] = cg_similarity_2D(l,la,xa)

DESCRIPTION 

% determines constraint values and Jacobians for 2D similarity

 Model:
 cg(x,l) = zr - s*zl - t = 0

 [l(3)]    - [x(1) -x(2)] [l(1)] - [x(3)] = [0]
 [l(4)]    - [x(2)  x(1)] [l(2)] - [x(4)]   [0]

 l         = 4x1-vector [xl yl,xr yr] = [zl, zr]
 x         = 4x1-vector [ a  b, c  d] = [ s,  t]
